
DROP TABLE Zeppline_feedback CASCADE CONSTRAINTS;

CREATE TABLE Zeppline_feedback
	(
	FEEDBACKID        NUMBER (5) NOT NULL,
	SEATCOMFORT       NUMBER (5) NOT NULL,
	CABINSTAFFSERVICE NUMBER (5) NOT NULL,
	FOODANDBEVERAGES  NUMBER (5) NOT NULL,
	INFLIGHTENT       NUMBER (5) NOT NULL,
	GROUNDSERVICES    NUMBER (5) NOT NULL,
	COMMENTS          VARCHAR2 (350) NOT NULL,
	USERID            NUMBER,
	PRIMARY KEY (FEEDBACKID),
	FOREIGN KEY (USERID) REFERENCES Zeppline_users (USERID)
	)
	

CREATE TABLE Zeppline_users
	(
	USERID    NUMBER NOT NULL,
	EMAILID   VARCHAR2 (30) NOT NULL,
	PASSWORD  VARCHAR2 (30) NOT NULL,
	FIRSTNAME VARCHAR2 (30) NOT NULL,
	LASTNAME  VARCHAR2 (30) NOT NULL,
	ADDRESS   VARCHAR2 (250) NOT NULL,
	COUNTRY   VARCHAR2 (30) NOT NULL,
	GENDER    VARCHAR2 (20) NOT NULL,
	STATUS    VARCHAR2 (4),
	IMAGEPATH VARCHAR2 (350),
	PRIMARY KEY (USERID),
	UNIQUE (EMAILID)
	)
   

CREATE SEQUENCE Zeppline_users_sequence 
START WITH 1
INCREMENT BY 1
